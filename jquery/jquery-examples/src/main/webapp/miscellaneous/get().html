.get()


This is summary of the function .get(), to more information see: .get() Documentation 


$(function(){
	
	var divs = $('#container > div').get().reverse();
	
	var contents = [];
	
	for ( var i = 0; i < divs.length; i++)
		 contents.push(divs[i].innerHTML);

	$('span').text( contents.join(" "));
	
});
