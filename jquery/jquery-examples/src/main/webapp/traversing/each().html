.each()


This is summary of the .each() function, to more information see: .each() Documentation 


$( function(){
	$('p', "#example-1").each(function(index) {
		$(this).text( "My number is = " + index );
	});		
});


$( function(){
	$('p', "#example-2").each(function(index) {	
		if( index % 2 == 0 )
			$(this).text( " I'm even paragraph " );			
		else
			$(this).text( " I'm odd paragraph " );
	});		
});
